# PDF para Excel Updater - Dependências Consolidadas (Múltiplas Versões)
# ========================================================================

# ===== DEPENDÊNCIAS CORE (OBRIGATÓRIAS - TODAS AS VERSÕES) =====
# Necessárias para processamento básico e linha de comando

pandas>=1.5.0              # Manipulação de dados e Excel
openpyxl>=3.0.0            # Leitura/escrita de arquivos Excel (.xlsx/.xlsm)
pdfplumber>=0.7.0          # Extração de texto de PDFs
python-dotenv>=1.0.0       # Carregamento de configurações .env

# ===== DEPENDÊNCIAS GUI v4.0 (PYQT6) =====
# Para interface gráfica moderna v4.0

PyQt6>=6.4.0               # Interface gráfica nativa e performática v4.0
                           # Requer Python 3.8+

# ===== DEPENDÊNCIAS GUI v3.x (CUSTOMTKINTER) =====
# Para interface gráfica v3.x (compatibilidade durante desenvolvimento)

customtkinter>=5.0.0       # Interface gráfica v3.x (pode coexistir com PyQt6)
pillow>=9.0.0              # Manipulação de imagens (requerido pelo CustomTkinter)
tkinterdnd2>=0.3.0         # Drag & Drop v3.x (opcional, pode coexistir)

# ===== DEPENDÊNCIAS DE BUILD =====
# Para geração de executável (opcional)

pyinstaller>=5.0.0         # Geração de executável standalone

# ===== NOTAS DE COMPATIBILIDADE =====
#
# COEXISTÊNCIA v3.x e v4.0:
# - Todas as dependências podem coexistir durante desenvolvimento
# - v3.x: python desktop_app.py (se disponível) usa CustomTkinter
# - v4.0: python desktop_app.py usa PyQt6
# - CLI: python pdf_to_excel_updater.py (funciona em ambas)
#
# Python Requirements:
# - v3.x: Python 3.7+ 
# - v4.0: Python 3.8+ (requerido para PyQt6)
#
# Windows: Todas as dependências funcionam nativamente
# macOS: Todas as dependências funcionam nativamente  
# Linux: Pode precisar de pacotes do sistema para PyQt6
#   Ubuntu/Debian: sudo apt-get install python3-pyqt6
#   CentOS/RHEL: sudo yum install python3-qt6
#   Arch: sudo pacman -S python-pyqt6
#
# ===== INSTALAÇÃO =====
#
# Opção 1 - Automática (Windows):
#   setup.bat  # Instala todas as dependências (v3.x + v4.0)
#
# Opção 2 - Manual (ambas versões):
#   pip install -r requirements.txt
#
# Opção 3 - Apenas v4.0 (PyQt6):
#   pip install pandas openpyxl pdfplumber python-dotenv PyQt6
#
# Opção 4 - Apenas v3.x (CustomTkinter):
#   pip install pandas openpyxl pdfplumber python-dotenv customtkinter pillow tkinterdnd2
#
# Opção 5 - Sem GUI (apenas linha de comando):
#   pip install pandas openpyxl pdfplumber python-dotenv
#
# ===== VERIFICAÇÃO =====
#
# Teste dependências core:
#   python -c "import pandas, openpyxl, pdfplumber, dotenv; print('Core: OK')"
#
# Teste v4.0 (PyQt6):
#   python -c "import PyQt6.QtWidgets; print('PyQt6: OK')"
#
# Teste v3.x (CustomTkinter):
#   python -c "import customtkinter; print('CustomTkinter: OK')"
#
# Teste interface específica:
#   python desktop_app.py  # Usa a versão disponível no código atual
